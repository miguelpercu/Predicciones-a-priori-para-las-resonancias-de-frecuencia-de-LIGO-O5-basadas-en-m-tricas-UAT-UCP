F_MIN = 2.0
F_MAX = 4000.0
F_STEP = 0.005
SEGMENT_WIDTH = 50.0

# Parámetros teóricos
ALPHA = 0.046
HANDSHAKE_PHASE = 7.07
NATURAL_PHASE = 70.39
TARGET_PHASE_DIFF = NATURAL_PHASE - HANDSHAKE_PHASE
TARGET_HARMONIC_DIFF = 2 * TARGET_PHASE_DIFF

TOL_HANDSHAKE = 5.5
TOL_NATURAL = 8.0
TOL_DIFF = 10.0
TOL_HARMONIC = 12.0

COH_THRESHOLD = 0.9
WINDOW_DUR = 128.0

# Tolerancia para bloques de fase
PHASE_TOLERANCE_BLOCK = 15.0   # grados

archivos = [

def cargar_ventana_segura(h1_path, l1_path, duracion):
    try:
        with h5py.File(h1_path, 'r') as f_h, h5py.File(l1_path, 'r') as f_l:
            h1_full = f_h['strain/Strain'][:]
            l1_full = f_l['strain/Strain'][:]
    except Exception as e:
        print(f"   Error cargando archivos: {e}")
        return None, None, None

    n_needed = int(duracion * FS)
    n_total = min(len(h1_full), len(l1_full))
    valido = ~(np.isnan(h1_full) | np.isnan(l1_full))

    inicio = -1
    count = 0
    for i in range(n_total):
        if valido[i]:
            if count == 0:
                inicio_cand = i
            count += 1
            if count >= n_needed:
                inicio = inicio_cand
                break
        else:
            count = 0

    if inicio == -1:
        print(f"   No se encontró una región válida de {duracion} s.")
        return None, None, None

    fin = inicio + n_needed
    h1_win = h1_full[inicio:fin]
    l1_win = l1_full[inicio:fin]
    return np.nan_to_num(h1_win), np.nan_to_num(l1_win), inicio/FS

def coherencia_rotacional_optimizada(fft_h1, fft_l1, freq, freqs_fft):
    idx = np.argmin(np.abs(freqs_fft - freq))
    v_h1 = fft_h1[idx]
    v_l1 = fft_l1[idx]
    max_coh = 0.0
    best_angle = 0.0
    for phi in np.linspace(0, 2*np.pi, 8, endpoint=False):
        v_l1_rot = v_l1 * np.exp(1j * phi)
        mag_h1 = np.abs(v_h1)
        mag_l1 = np.abs(v_l1_rot)
        if mag_h1 > 1e-60 and mag_l1 > 1e-60:
            coh = np.abs(np.vdot(v_h1, v_l1_rot)) / (mag_h1 * mag_l1)
            if coh > max_coh:
                max_coh = coh
                best_angle = phi
    return max_coh, best_angle

def analizar_banda_optimizado(h1_win, l1_win, f_min, f_max, f_step):
    fft_h1 = np.fft.rfft(h1_win)
    fft_l1 = np.fft.rfft(l1_win)
    freqs_fft = np.fft.rfftfreq(len(h1_win), 1/FS)

    f_min_eff = max(f_min, freqs_fft[0])
    f_max_eff = min(f_max, freqs_fft[-1])
    freqs = np.arange(f_min_eff, f_max_eff, f_step)

    resultados = []
    for f in freqs:
        coh, ang = coherencia_rotacional_optimizada(fft_h1, fft_l1, f, freqs_fft)
        if coh > COH_THRESHOLD:
            idx = np.argmin(np.abs(freqs_fft - f))
            v_h1 = fft_h1[idx]
            v_l1 = fft_l1[idx]
            v_l1_opt = v_l1 * np.exp(1j * ang)
            phase = np.angle(v_h1 * np.conj(v_l1_opt), deg=True) % 360
            resultados.append({
                'freq': f,
                'coh': coh,
                'phase': phase,
                'ang_opt': ang
            })
    return resultados

def detectar_bloques_fase(resultados, delta_f_max=0.005, phase_tol=PHASE_TOLERANCE_BLOCK):
    if not resultados:
        return []
    resultados.sort(key=lambda x: x['freq'])
    bloques = []
    bloque_actual = {'freqs': [], 'phases': [], 'cohs': []}
    for r in resultados:
        if len(bloque_actual['freqs']) == 0:
            bloque_actual['freqs'].append(r['freq'])
            bloque_actual['phases'].append(r['phase'])
            bloque_actual['cohs'].append(r['coh'])
        else:
            ult_freq = bloque_actual['freqs'][-1]
            if (r['freq'] - ult_freq) < delta_f_max and abs(r['phase'] - np.mean(bloque_actual['phases'])) <= phase_tol:
                bloque_actual['freqs'].append(r['freq'])
                bloque_actual['phases'].append(r['phase'])
                bloque_actual['cohs'].append(r['coh'])
            else:
                if len(bloque_actual['freqs']) >= 3:
                    bloques.append({
                        'f_min': bloque_actual['freqs'][0],
                        'f_max': bloque_actual['freqs'][-1],
                        'phase_mean': np.mean(bloque_actual['phases']),
                        'phase_std': np.std(bloque_actual['phases']),
                        'frecuencias': bloque_actual['freqs'],
                        'coherencias': bloque_actual['cohs']
                    })
                bloque_actual = {'freqs': [r['freq']], 'phases': [r['phase']], 'cohs': [r['coh']]}
    if len(bloque_actual['freqs']) >= 3:
        bloques.append({
            'f_min': bloque_actual['freqs'][0],
            'f_max': bloque_actual['freqs'][-1],
            'phase_mean': np.mean(bloque_actual['phases']),
            'phase_std': np.std(bloque_actual['phases']),
            'frecuencias': bloque_actual['freqs'],
            'coherencias': bloque_actual['cohs']
        })
    return bloques

def buscar_pares_alfa(candidatos, alpha=ALPHA, tol=0.005):
    pares = []
    n = len(candidatos)
    for i in range(n):
        for j in range(i+1, n):
            df = abs(candidatos[i]['freq'] - candidatos[j]['freq'])
            if abs(df - alpha) <= tol:
                diff_phase = abs(candidatos[i]['phase'] - candidatos[j]['phase']) % 360
                diff_phase = min(diff_phase, 360 - diff_phase)
                if abs(diff_phase - TARGET_PHASE_DIFF) <= TOL_DIFF:
                    tipo = "Primaria"
                elif abs(diff_phase - TARGET_HARMONIC_DIFF) <= TOL_HARMONIC:
                    tipo = "Armónica"
                else:
                    tipo = "Sin clasificar"
                pares.append({
                    'f1': candidatos[i]['freq'],
                    'f2': candidatos[j]['freq'],
                    'df': df,
                    'phase1': candidatos[i]['phase'],
                    'phase2': candidatos[j]['phase'],
                    'diff_phase': diff_phase,
                    'tipo': tipo
                })
    return pares

def guardar_segmento(candidatos, archivo_nombre, gps, f_min, f_max):
    if not candidatos:
        return False
    df = pd.DataFrame(candidatos)
    filename = f"RH_{archivo_nombre}_{gps}_{f_min:.1f}-{f_max:.1f}Hz.csv"
    filepath = os.path.join(OUTPUT_DIR, filename)
    df.to_csv(filepath, index=False)
    print(f"   Guardado: {filename}")
    # Liberar memoria del DataFrame local
    del df
    return True

def flush_drive():
    try:
        os.sync()
        time.sleep(0.5)
        print("   [Sistema] Sincronización de archivos forzada.")
    except Exception as e:
        print(f"   [Advertencia] No se pudo sincronizar: {e}")

def log_progreso(archivo_nombre, gps, f_min, f_max, lineas, bloques, pares, timestamp):
    log_path = os.path.join(OUTPUT_DIR, "RH_v8.7_progress.log")
    with open(log_path, "a") as f:
        f.write(f"{timestamp} | {archivo_nombre} | GPS {gps} | "
                f"{f_min:.1f}-{f_max:.1f} Hz | líneas={lineas} | bloques={bloques} | pares={pares}\n")

    h1_win, l1_win, t0 = cargar_ventana_segura(arch['h1'], arch['l1'], WINDOW_DUR)
    if h1_win is None:
        print("   No se pudo cargar la ventana. Saltando.")
        continue

    gps_procesados.add(arch['gps'])

    f_inicio = F_MIN
    while f_inicio < F_MAX:
        f_fin = min(f_inicio + SEGMENT_WIDTH, F_MAX)
        print(f"\n   Segmento: {f_inicio:.1f}–{f_fin:.1f} Hz")

        filename_pattern = f"RH_{arch['nombre']}_{arch['gps']}_{f_inicio:.1f}-{f_fin:.1f}Hz.csv"
        filepath = os.path.join(OUTPUT_DIR, filename_pattern)
        if os.path.exists(filepath):
            print(f"   Archivo ya existe, omitiendo segmento.")
            f_inicio = f_fin
            continue

        t_start = time.time()
        resultados = analizar_banda_optimizado(h1_win, l1_win, f_inicio, f_fin, F_STEP)
        t_end = time.time()
        print(f"   Procesado en {t_end - t_start:.1f} s. Líneas encontradas: {len(resultados)}")

        for r in resultados:
            r.update({'archivo': arch['nombre'], 'gps': arch['gps'], 't_inicio': t0})

        bloques = detectar_bloques_fase(resultados)
        pares = buscar_pares_alfa(resultados)
        print(f"   Bloques contiguos: {len(bloques)}")
        print(f"   Pares con Δf ≈ {ALPHA} Hz: {len(pares)}")

        if resultados:
            guardar_segmento(resultados, arch['nombre'], arch['gps'], f_inicio, f_fin)

        flush_drive()
        log_progreso(arch['nombre'], arch['gps'], f_inicio, f_fin,
                     len(resultados), len(bloques), len(pares),
                     datetime.now().strftime("%Y-%m-%d %H:%M:%S"))

        # Liberar memoria después de cada segmento
        del resultados, bloques, pares
        gc.collect()
        print("   [Sistema] Memoria liberada para el siguiente segmento.")

        f_inicio = f_fin

    # Liberar memoria de la ventana grande al terminar el archivo
    del h1_win, l1_win, t0
    gc.collect()
    print(f"   [Sistema] Memoria liberada para el archivo {arch['nombre']}.")

print("\n" + "="*80)
print("✅ ANÁLISIS BLOQUE A COMPLETADO")
print(f"   GPS procesados: {sorted(gps_procesados)}")
print("   Resultados guardados en /content/drive/MyDrive/UAT_Resultados/")
print("="*80)

def extraer_rango(nombre_archivo):
    """Extrae el rango de frecuencias del nombre del archivo."""
    match = re.search(r'(\d+\.?\d*)-(\d+\.?\d*)Hz', nombre_archivo)
    if match:
        return float(match.group(1)), float(match.group(2))
    return None, None

def detectar_bloques_fase(resultados, phase_tol=PHASE_TOLERANCE_BLOCK, delta_f_max=0.005):
    """
    Detecta bloques de fase contiguos a partir de una lista de resultados.
    Cada resultado debe tener 'freq' y 'phase'.
    """
    if not resultados:
        return []
    resultados.sort(key=lambda x: x['freq'])
    bloques = []
    bloque_actual = {'freqs': [], 'phases': []}

    for r in resultados:
        if len(bloque_actual['freqs']) == 0:
            bloque_actual['freqs'].append(r['freq'])
            bloque_actual['phases'].append(r['phase'])
        else:
            ult_freq = bloque_actual['freqs'][-1]
            if (r['freq'] - ult_freq) < delta_f_max and abs(r['phase'] - np.mean(bloque_actual['phases'])) <= phase_tol:
                bloque_actual['freqs'].append(r['freq'])
                bloque_actual['phases'].append(r['phase'])
            else:
                if len(bloque_actual['freqs']) >= 3:
                    bloques.append({
                        'f_min': bloque_actual['freqs'][0],
                        'f_max': bloque_actual['freqs'][-1],
                        'phase_mean': np.mean(bloque_actual['phases']),
                        'phase_std': np.std(bloque_actual['phases']),
                        'frecuencias': bloque_actual['freqs']
                    })
                bloque_actual = {'freqs': [r['freq']], 'phases': [r['phase']]}

    if len(bloque_actual['freqs']) >= 3:
        bloques.append({
            'f_min': bloque_actual['freqs'][0],
            'f_max': bloque_actual['freqs'][-1],
            'phase_mean': np.mean(bloque_actual['phases']),
            'phase_std': np.std(bloque_actual['phases']),
            'frecuencias': bloque_actual['freqs']
        })
    return bloques

def buscar_pares_alfa_eficiente(frecuencias, phases, alpha=ALPHA, tol=0.005):
    """
    Busca pares de frecuencias con diferencia ≈ alpha de manera eficiente.
    Recibe arrays ordenados de frecuencias y sus fases asociadas.
    Retorna lista de pares con información.
    """
    pares = []
    n = len(frecuencias)
    j = 0
    for i in range(n):
        objetivo = frecuencias[i] + alpha
        while j < n and frecuencias[j] < objetivo - tol:
            j += 1
        for k in range(j, min(n, j+10)):
            df = frecuencias[k] - frecuencias[i]
            if abs(df - alpha) <= tol:
                diff_phase = abs(phases[i] - phases[k]) % 360
                diff_phase = min(diff_phase, 360 - diff_phase)
                if abs(diff_phase - TARGET_PHASE_DIFF) <= TOL_DIFF:
                    tipo = "Primaria"
                elif abs(diff_phase - TARGET_HARMONIC_DIFF) <= TOL_HARMONIC:
                    tipo = "Armónica"
                else:
                    tipo = "Sin clasificar"
                pares.append({
                    'f1': frecuencias[i],
                    'f2': frecuencias[k],
                    'df': df,
                    'phase1': phases[i],
                    'phase2': phases[k],
                    'diff_phase': diff_phase,
                    'tipo': tipo
                })
            elif frecuencias[k] > objetivo + tol:
                break
    return pares

- Filtra pares con persistencia ≥ 75% (nodos de acero).
- Realiza micro‑zoom en 359.912 Hz.

import numpy as np
import pandas as pd
import os
from collections import defaultdict
import warnings
warnings.filterwarnings('ignore')

# =============================================================================
# CONFIGURACIÓN
# =============================================================================
DATA_DIR = "/content/drive/MyDrive/UAT_Resultados/"   # <-- AJUSTA SI ES NECESARIO

OUTPUT_ZOOM = os.path.join(DATA_DIR, "zoom_359.912Hz.csv")

ALPHA = 0.046                    # Hz
TOL_ALPHA = 0.001                # Tolerancia para Δf
PRIMARY_PHASE = 63.32            # °
HARMONIC_PHASE = 130.6           # °
PHASE_TOL = 7.0                  # °
PERSISTENCE_FRACTION = 0.75      # 75%
PREDICTED_FREQ = 359.912         # Hz
PREDICTED_TOL = 0.01             # Hz

# =============================================================================
# FUNCIONES
# =============================================================================
def buscar_pares_en_grupo(frecuencias, fases):
    """
    Busca pares dentro de un grupo (una ventana GPS) con Δf ≈ ALPHA.
    Retorna lista de tuplas (f1_round, f2_round, diff_round) para cada par.
    """
    pares = []
    n = len(frecuencias)
    # Ordenar (ya deberían estar ordenadas, pero por seguridad)
    idx = np.argsort(frecuencias)
    frecuencias = frecuencias[idx]
    fases = fases[idx]

    j = 0
    for i in range(n):
        objetivo = frecuencias[i] + ALPHA
        while j < n and frecuencias[j] < objetivo - TOL_ALPHA:
            j += 1
        for k in range(j, min(n, j+20)):
            df = frecuencias[k] - frecuencias[i]
            if abs(df - ALPHA) <= TOL_ALPHA:
                diff = abs(fases[i] - fases[k]) % 360
                diff = min(diff, 360 - diff)
                # Redondeamos para agrupar de manera estable
                f1_round = round(frecuencias[i], 6)
                f2_round = round(frecuencias[k], 6)
                diff_round = round(diff, 2)
                pares.append((f1_round, f2_round, diff_round))
            elif frecuencias[k] > objetivo + TOL_ALPHA:
                break
    return pares

def clasificar_par(diff_phase):
    """Clasifica según diferencia de fase."""
    if abs(diff_phase - PRIMARY_PHASE) <= PHASE_TOL:
        return 'primaria'
    elif abs(diff_phase - HARMONIC_PHASE) <= PHASE_TOL:
        return 'armonica'
    else:
        return 'sin_clasificar'

# =============================================================================
# EJECUCIÓN PRINCIPAL
# =============================================================================

print("RESONANT HUNTER v8.8 – PERSISTENCIA Y MICRO‑ZOOM (MEMORIA OPTIMIZADA)")

})

# 2. Procesar cada GPS y acumular conteo de pares
print("\nProcesando cada GPS para encontrar pares...")
contador_pares = defaultdict(set)   # (f1, f2, diff_round) -> set de GPS

for gps in gps_unicos:
    grupo = df[df['gps'] == gps]
    freqs = grupo['freq'].values
    phases = grupo['phase'].values
    pares = buscar_pares_en_grupo(freqs, phases)
    for p in pares:
        contador_pares[p].add(gps)
    print(f"  GPS {gps}: {len(pares)} pares encontrados.")

# 3. Construir DataFrame con persistencia
print("\nCalculando persistencia...")
datos = []
for (f1, f2, diff), gps_set in contador_pares.items():
    ventanas = len(gps_set)
    persistencia = ventanas / n_gps
    datos.append({
        'f1': f1,
        'f2': f2,
        'diff_phase': diff,
        'ventanas': ventanas,
        'persistencia': persistencia
    })

df_persistencia = pd.DataFrame(datos)
print(f"  Total de pares únicos (en todas las ventanas): {len(df_persistencia):,}")

# 4. Filtrar nodos de acero (persistencia >= 75%)
df_acero = df_persistencia[df_persistencia['persistencia'] >= PERSISTENCE_FRACTION].copy()
print(f"Nodos de acero encontrados: {len(df_acero)}")

if len(df_acero) > 0:
    # Agregar clasificación
    df_acero['tipo'] = df_acero['diff_phase'].apply(clasificar_par)
    df_acero['f_media'] = (df_acero['f1'] + df_acero['f2']) / 2
    # Guardar
    df_acero.to_csv(OUTPUT_ELITE, index=False)
    print(f"  CSV de élite guardado en {OUTPUT_ELITE}")

    # 5. Micro‑zoom en 359.912 Hz
    print(f"\nMicro‑zoom en frecuencia predictiva {PREDICTED_FREQ} Hz...")
    zoom = df_acero[(df_acero['f_media'] >= PREDICTED_FREQ - PREDICTED_TOL) &
                    (df_acero['f_media'] <= PREDICTED_FREQ + PREDICTED_TOL)]
    print(f"  Pares en el entorno: {len(zoom)}")
    if len(zoom) > 0:
        print("  Ejemplos:")
        for _, row in zoom.head(10).iterrows():
            print(f"    f1={row['f1']:.6f}, f2={row['f2']:.6f}, Δφ={row['diff_phase']:.2f}°, tipo={row['tipo']}")
        zoom.to_csv(OUTPUT_ZOOM, index=False)
        print(f"  Archivo de zoom guardado en {OUTPUT_ZOOM}")
else:
    print("No se alcanzó el umbral de persistencia. Puedes reducir PERSISTENCE_FRACTION para explorar.")

- Clasifica según nodos de fase (múltiplos de 45°) con margen del 7%.
- Realiza micro‑zoom en 359.912 Hz.

"""

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import os
from collections import defaultdict
import warnings
warnings.filterwarnings('ignore')

# =============================================================================
# CONFIGURACIÓN
# =============================================================================
ALPHA = 0.046
TOL_ALPHA = 0.001
PERSISTENCE_FRACTION = 0.50

# Nodos de fase del protocolo Percudani (múltiplos de 45°)
PHASE_NODES = np.arange(0, 361, 45)      # 0°,45°,90°,135°,180°,225°,270°,315°,360°
PHASE_TOLERANCE = 25.2                   # 7% de 360°

PREDICTED_FREQ = 359.912
PREDICTED_TOL = 0.01

DATA_DIR = "/content/drive/MyDrive/UAT_Resultados/"
COMBINED_FILE = os.path.join(DATA_DIR, "todos_candidatos_combinado.csv")
OUTPUT_ELITE = os.path.join(DATA_DIR, "nodos_acero_UAT.csv")
OUTPUT_ZOOM = os.path.join(DATA_DIR, "zoom_359.912Hz.csv")

# =============================================================================
# FUNCIONES AUXILIARES
# =============================================================================
def buscar_pares_en_grupo(frecuencias, fases):
    """
    Busca pares con Δf ≈ ALPHA dentro de un grupo (un GPS).
    Retorna lista de tuplas (f1_round, f2_round, diff_round).
    """
    pares = []
    n = len(frecuencias)
    idx = np.argsort(frecuencias)
    frecuencias = frecuencias[idx]
    fases = fases[idx]

    j = 0
    for i in range(n):
        objetivo = frecuencias[i] + ALPHA
        while j < n and frecuencias[j] < objetivo - TOL_ALPHA:
            j += 1
        for k in range(j, min(n, j+20)):
            df = frecuencias[k] - frecuencias[i]
            if abs(df - ALPHA) <= TOL_ALPHA:
                diff = abs(fases[i] - fases[k]) % 360
                diff = min(diff, 360 - diff)
                # Redondeo para agrupación estable
                f1_round = round(frecuencias[i], 6)
                f2_round = round(frecuencias[k], 6)
                diff_round = round(diff, 2)
                pares.append((f1_round, f2_round, diff_round))
            elif frecuencias[k] > objetivo + TOL_ALPHA:
                break
    return pares

def clasificar_par_por_nodo(diff_phase, nodos, tolerancia):
    """Clasifica un par según el nodo más cercano dentro de la tolerancia."""
    distancias = [abs(diff_phase - n) for n in nodos]
    min_dist = min(distancias)
    if min_dist <= tolerancia:
        idx = distancias.index(min_dist)
        nodo = nodos[idx]
        if nodo == 360:
            nodo = 0
        return f'nodo_{nodo}°'
    else:
        return 'sin_clasificar'

def micro_zoom(frecuencia, pares_df, tol=PREDICTED_TOL):
    """Extrae pares en el entorno de una frecuencia predictiva."""
    mask = (np.abs(pares_df['f1'] - frecuencia) <= tol) | \
           (np.abs(pares_df['f2'] - frecuencia) <= tol)
    return pares_df[mask].copy()

# =============================================================================

contador_pares = defaultdict(set)   # (f1_round, f2_round, diff_round) -> set de GPS

for gps in gps_unicos:
    grupo = df[df['gps'] == gps]
    freqs = grupo['freq'].values
    phases = grupo['phase'].values
    pares = buscar_pares_en_grupo(freqs, phases)
    for p in pares:
        contador_pares[p].add(gps)
    print(f"   GPS {gps}: {len(pares)} pares encontrados.")

# 3. Construir DataFrame con persistencia
print("\nCalculando persistencia...")
datos = []
for (f1, f2, diff), gps_set in contador_pares.items():
    ventanas = len(gps_set)
    persistencia = ventanas / n_gps
    datos.append({
        'f1': f1,
        'f2': f2,
        'diff_phase': diff,
        'ventanas': ventanas,
        'persistencia': persistencia
    })

df_pares = pd.DataFrame(datos)
print(f"   Total de pares únicos (en todas las ventanas): {len(df_pares):,}")

# 4. Filtrar por persistencia (nodos de acero)

if len(df_acero) > 0:

    df_acero['tipo'] = df_acero['diff_phase'].apply(
        lambda x: clasificar_par_por_nodo(x, PHASE_NODES, PHASE_TOLERANCE)
    )
    df_acero['f_media'] = (df_acero['f1'] + df_acero['f2']) / 2

print(f"\nMicro‑zoom en frecuencia predictiva {PREDICTED_FREQ} Hz...")
df_zoom = micro_zoom(PREDICTED_FREQ, df_pares, tol=PREDICTED_TOL)

if len(df_zoom) > 0:
    # Clasificar también estos pares

    df_zoom['f_media'] = (df_zoom['f1'] + df_zoom['f2']) / 2
    # Mostrar ejemplos
    print("   Ejemplos de pares en la zona:")

        print(f"      {row['f1']:.3f} – {row['f2']:.3f} Hz | Δf = {abs(row['f1']-row['f2']):.3f} Hz | Δφ = {row['diff_phase']:.1f}° | {row['tipo']}")

def verificar_interferencia():
    # Creamos los 8 frentes de fase (0, 45, 90, 135, 180, 225, 270, 315)
    fases_frentes = np.radians(np.arange(0, 360, desfase_unitario))
    
    # En un sistema de interferencia rotacional constructiva (RMS ~0.707)
    # representamos cada frente como un vector unitario
    vectores_x = np.cos(fases_frentes)
    vectores_y = np.sin(fases_frentes)
    
    # Suma vectorial (Interferencia)
    sum_x = np.sum(vectores_x)
    sum_y = np.sum(vectores_y)
    
    # En un sistema perfectamente balanceado, la suma sería 0.
    # Pero la "Firma del Higo" es un pulso vital expansivo, 
    # buscamos el residuo de fase que queda en el detector.
    angulo_resultante_rad = np.arctan2(sum_y, sum_x)
    angulo_resultante_deg = np.degrees(angulo_resultante_rad) % 360
    
    print(f"--- Validación de Autoría Percudani ---")
    print(f"Fase Observada (GPS 4): {fase_observada_deg:.3f}°")
    print(f"Diferencia con el cierre de ciclo (360°): {360 - fase_observada_deg:.3f}°")
    
    # El error de calibración del 7% que mencionaste:
    error = (abs(360 - fase_observada_deg) / 360) * 100
    print(f"Margen de ruido térmico detectado: {error:.2f}%")
    
    if error <= 7.0:
        print("\n✅ RESULTADO: El error está dentro de tu margen de calibración (7%).")
        print("La fase en el GPS 4 confirma la 'congelación' de la Firma del Higo.")
    else:
        print("\n⚠️ RESULTADO: El error supera el 7%. Requiere ajuste de deriva (alpha).")

verificar_interferencia()

    pares = []

- Busca pares con Δf ≈ 0.046 Hz en cada GPS.
- Para cada par, acumula las coherencias de las dos frecuencias **solo si son >= COH_THRESHOLD**.

- Evalúa error respecto al target teórico √0.5·k_early.
- Genera estadísticas, gráficos y CSV de élite.
================================================================================

import numpy as np

from collections import defaultdict

warnings.filterwarnings('ignore')

# PARÁMETROS DEL PROTOCOLO
# =============================================================================
ALPHA = 0.046
TOL_ALPHA = 0.001
k_early = 0.967
RMS_target = np.sqrt(0.5) * k_early      # ≈ 0.683
IVANCHO_LIMIT = 4.978

# Umbral de persistencia (puedes cambiarlo)
PERSISTENCE_THRESHOLD = 0.50

# Margen de error para densidad energética y fase (7%)
TOLERANCE_RMS = 7.0                      # %
PHASE_TOLERANCE = 25.2                   # grados (7% de 360°)

# Umbral de coherencia para considerar una ocurrencia válida
COH_THRESHOLD = 0.9                      # solo coherencias >= este valor se usan en el RMS
RMS_METHOD = 'rms'                       # 'rms' o 'median'

# Nodos de fase cuántica (ajustados por k_early)
QUANTUM_STEP = 45.0 * k_early            # ≈ 43.515°
QUANTUM_NODES = np.arange(0, 360, QUANTUM_STEP)
QUANTUM_NODES = np.append(QUANTUM_NODES, 360)

PREDICTED_FREQ = 359.912
PREDICTED_TOL = 0.01

COMBINED_FILE = os.path.join(DATA_DIR, "todos_candidatos_combinado.csv")

SCATTER_FIG = os.path.join(DATA_DIR, "scatter_persistence_error.png")

def buscar_pares_en_grupo(frecuencias, fases, cohs):
    """
    Busca pares con Δf ≈ ALPHA dentro de un grupo (un GPS).
    Retorna lista de tuplas (f1_round, f2_round, diff_round, coh1, coh2)
    """
    pares = []
    n = len(frecuencias)
    idx = np.argsort(frecuencias)
    frecuencias = frecuencias[idx]
    fases = fases[idx]
    cohs = cohs[idx]

    j = 0
    for i in range(n):
        objetivo = frecuencias[i] + ALPHA
        while j < n and frecuencias[j] < objetivo - TOL_ALPHA:
            j += 1
        for k in range(j, min(n, j+20)):
            df = frecuencias[k] - frecuencias[i]
            if abs(df - ALPHA) <= TOL_ALPHA:
                diff = abs(fases[i] - fases[k]) % 360
                diff = min(diff, 360 - diff)
                f1_round = round(frecuencias[i], 6)
                f2_round = round(frecuencias[k], 6)
                diff_round = round(diff, 2)
                pares.append((f1_round, f2_round, diff_round, cohs[i], cohs[k]))
            elif frecuencias[k] > objetivo + TOL_ALPHA:
                break
    return pares

def nearest_quantum_node(phase):
    """Distancia al nodo cuántico más cercano."""
    return min(abs(phase - n) for n in QUANTUM_NODES)

def classify_quantum_phase(phase):
    dist = nearest_quantum_node(phase)
    if dist <= PHASE_TOLERANCE:
        idx = np.argmin([abs(phase - n) for n in QUANTUM_NODES])
        node = QUANTUM_NODES[idx]
        if node == 360:
            node = 0.0
        return f"nodo_{node:.2f}°"
    else:
        return "sin_clasificar"

def compute_rms_filtered(cohs_list):
    """Calcula el RMS de la lista de coherencias filtrada por COH_THRESHOLD."""
    filtered = [c for c in cohs_list if c >= COH_THRESHOLD]
    if not filtered:
        return 0.0
    if RMS_METHOD == 'rms':
        return np.sqrt(np.mean(np.array(filtered)**2))
    elif RMS_METHOD == 'median':
        return np.median(filtered)
    else:
        return np.mean(filtered)

def classify_status(k_error, upc_ratio):
    if upc_ratio > 4.5:
        return "Overdrive"
    elif k_error < TOLERANCE_RMS:
        return "Estable"
    else:
        return "Fuga Térmica"

def micro_zoom(frecuencia, df_pares, tol=PREDICTED_TOL):
    mask = (np.abs(df_pares['f1'] - frecuencia) <= tol) | \
           (np.abs(df_pares['f2'] - frecuencia) <= tol)
    return df_pares[mask].copy()
